## Single function to obtain the statistical performance measures

def get_stat_performance(df):

    # mean bias
    
    mb_pblh = (df['model_pbl'] - df['pbl_stull_hgt']).describe()[1]
    
    print("Mean bias :")
    print("mean bias pblh = ", mb_pblh, "\n")
    
    fb_pblh = 2 * ((df['pbl_stull_hgt'].describe()[1] - df['model_pbl'].describe()[1]) / \
                   (df['pbl_stull_hgt'].describe()[1] + df['model_pbl'].describe()[1]))
    
    print("fractional bias is :")
    print("fractional bias pblh = ", fb_pblh, "\n")
    
    r_pblh = ((df['pbl_stull_hgt'] - df['pbl_stull_hgt'].describe()[1]) * \
              (df['model_pbl'] - df['model_pbl'].describe()[1])).describe()[1] / (df['model_pbl'].describe()[2] * \
                                                                                 df['pbl_stull_hgt'].describe()[2])
    print("Correlation coefficient is :")
    print("corr coeff pblh = ", r_pblh)
